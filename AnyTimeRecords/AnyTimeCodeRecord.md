#### 两个变量交换值,不使用第三个变量
    a = 10, b = 20交换变量值
    a = a + b
    b = a - b // b = a + b - b => b = a //此时b = a
    a = a - b // a = a + b - a => a = b
#### 取模公式 a % b = a - a / b * b
    10 % 3 = 1
    -10 % 3 = -1 // -10 - (-10) / 3 * 3 = -10 - (-9) = -1
    10 % -3 = 1 // 10 - 10 / -3 * -3 = 1
    -10 % -3 -1 // -10 - (-10) / (-3) * (-3) = -1
#### 字母大小写位运算转换
    大写变小写、小写变大写：字符 ^= 32
    大写变小写、小写变小写：字符 |= 32
    大写变大写、小写变大写：字符 &= 33
